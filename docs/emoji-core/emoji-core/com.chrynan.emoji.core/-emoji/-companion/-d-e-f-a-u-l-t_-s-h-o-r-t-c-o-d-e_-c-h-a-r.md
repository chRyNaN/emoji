# DEFAULT_SHORTCODE_CHAR

[common]\
const val [DEFAULT_SHORTCODE_CHAR](-d-e-f-a-u-l-t_-s-h-o-r-t-c-o-d-e_-c-h-a-r.md): [Char](https://kotlinlang.org/api/latest/jvm/stdlib/kotlin/-char/index.html) = ':'
